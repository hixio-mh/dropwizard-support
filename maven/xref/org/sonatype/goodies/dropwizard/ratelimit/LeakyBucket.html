13  package org.sonatype.goodies.dropwizard.ratelimit;
14  
15  import com.google.common.base.MoreObjects;
16  import com.google.common.base.Ticker;
17  import io.dropwizard.util.Duration;
18  import org.slf4j.Logger;
19  import org.slf4j.LoggerFactory;
20  
21  import static com.google.common.base.Preconditions.checkArgument;
22  import static com.google.common.base.Preconditions.checkNotNull;
23  
24  /**
25   * Leaky bucket algorithm.
26   *
27   * Bucket starts out empty with a {@link #capacity}, and a drain {@link #drainCountPerPeriod count-per-period} and
28   * {@link #drainPeriodNanos period}.
29   *
30   * Drops (units of work or requests or whatever) are added to the bucket.
31   * Before drops are added, the bucket is potentially drained based on configured rate.
32   *
33   * If the bucket is ever filled with more drops than {@link #capacity} then the bucket is considered overflown.
34   * Overflown drops are all discarded, even if a portion of drops could fit before overflowing with {@link #fill(long)}.
35   *
36   * @since 1.2.0
37   */
38  public class LeakyBucket
39  {
40    private static final Logger log = LoggerFactory.getLogger(LeakyBucket.class);
41  
42    /**
43     * Ticker to track time.
44     */
45    private final Ticker ticker;
46  
47    /**
48     * The total capacity of the bucket.
49     */
50    private final long capacity;
51  
52    /**
53     * How many drops to drain per-period.
54     *
55     * @see #drainPeriodNanos
56     */
57    private final long drainCountPerPeriod;
58  
59    /**
60     * The period at which drops drain in nano-seconds.
61     */
62    private final long drainPeriodNanos;
63  
64    /**
65     * The current number of drops in the bucket.
66     */
67    private long size = 0;
68  
69    /**
70     * The time the bucket was last drained.
71     */
72    private long lastDrainTime;
73  
74    /**
75     * The next time the bucket may be drained.
76     */
77    private long nextDrainTime;
78  
79    public LeakyBucket(final Ticker ticker,
80                       final long capacity,
81                       final long drainCountPerPeriod,
82                       final Duration drainPeriod)
83    {
84      this.ticker = checkNotNull(ticker);
85      checkArgument(capacity > 0);
86      this.capacity = capacity;
87      checkArgument(drainCountPerPeriod > 0);
88      this.drainCountPerPeriod = drainCountPerPeriod;
89      this.drainPeriodNanos = checkNotNull(drainPeriod).toNanoseconds();
90  
91      // set initial last+next drain-times
92      long now = ticker.read();
93      lastDrainTime = now;
94      nextDrainTime = now + drainPeriodNanos;
95    }
96  
97    /**
98     * Returns bucket capacity.
99     */
100   public long getCapacity() {
101     return capacity;
102   }
103 
104   /**
105    * Get current size (number of drops) in bucket.
106    */
107   public long getSize() {
108     return size;
109   }
110 
111   /**
112    * Attempt to drain bucket based on constant rate expressed by {@link #drainCountPerPeriod} and {@link #drainPeriodNanos}.
113    *
114    * @return Drops drained.
115    */
116   public long drain() {
117     long now = ticker.read();
118 
119     // skip if not ready to drain yet
120     if (now < nextDrainTime) {
121       return 0;
122     }
123 
124     // calculate how many periods have been missed
125     long missedPeriods = Math.max(0, (now - lastDrainTime) / drainPeriodNanos);
126 
127     // adjust last drain-time; calculate the target time based on now many periods we missed since last drained
128     lastDrainTime += missedPeriods * drainPeriodNanos;
129 
130     // adjust next drain-time
131     nextDrainTime = lastDrainTime + drainPeriodNanos;
132 
133     // calculate how many to drain (adjusted for max available to drain)
134     long drainCount = Math.min(size, missedPeriods * drainCountPerPeriod);
135     size -= drainCount;
136     log.trace("Drained: {}; new-size: {}", drainCount, size);
137 
138     return drainCount;
139   }
140 
141   /**
142    * Attempt to fill bucket.
143    *
144    * Before filling bucket is checked for potential drainage.
145    *
146    * @param count   Number of drops to add to bucket.
147    * @return        {@code true} if adding count drops to bucket would overflow.
148    *                All (including non-overflowing) drops are discarded.
149    */
150   public boolean fill(final long count) {
151     checkArgument(count > 0);
152 
153     // first attempt to drain
154     drain();
155 
156     // calculate the target updated size with count drops
157     long updated = size + count;
158     if (updated > capacity) {
159       // bucket would overflow if count drops were added
160       return true;
161     }
162 
163     log.trace("Fill: {}", updated);
164     size = updated;
165     return false;
166   }
167 
168   /**
169    * Attempt to fill bucket by one.
170    *
171    * @see #fill(long)
172    */
173   public boolean fill() {
174     return fill(1);
175   }
176 
177   @Override
178   public String toString() {
179     return MoreObjects.toStringHelper(this)
180         .add("capacity", capacity)
181         .add("drainCountPerPeriod", drainCountPerPeriod)
182         .add("drainPeriodNanos", drainPeriodNanos)
183         .add("size", size)
184         .add("lastDrainTime", lastDrainTime)
185         .add("nextDrainTime", nextDrainTime)
186         .toString();
187   }
188 }
