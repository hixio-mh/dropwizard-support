13  package org.sonatype.goodies.dropwizard.events;
14  
15  import java.lang.reflect.Constructor;
16  import java.lang.reflect.Method;
17  import java.util.concurrent.Executor;
18  
19  import javax.inject.Named;
20  import javax.inject.Singleton;
21  
22  import com.google.common.base.Throwables;
23  import com.google.common.eventbus.EventBus;
24  import com.google.common.eventbus.SubscriberExceptionHandler;
25  import org.slf4j.LoggerFactory;
26  
27  import static com.google.common.base.Preconditions.checkNotNull;
28  import static com.google.common.util.concurrent.MoreExecutors.directExecutor;
29  
30  // see: https://github.com/sonatype/nexus-public/blob/master/components/nexus-common/src/main/java/org/sonatype/nexus/common/event/EventBusFactory.java
31  
32  /**
33   * Default {@link EventBusFactory}.
34   *
35   * @since 1.2.0
36   */
37  @Named
38  @Singleton
39  public class EventBusFactoryImpl
40    implements EventBusFactory
41  {
42    @Override
43    public EventBus create(final String name) {
44      checkNotNull(name);
45      return newEventBus(name, directExecutor());
46    }
47  
48    @Override
49    public EventBus create(final String name, final Executor executor) {
50      checkNotNull(name);
51      checkNotNull(executor);
52      return newEventBus(name, executor);
53    }
54  
55    private static EventBus newEventBus(final String name, final Executor executor) {
56      try {
57        Class<?> dispatcherClass = EventBus.class.getClassLoader().loadClass("com.google.common.eventbus.Dispatcher");
58  
59        // immediate dispatcher means events are always processed in a reentrant fashion
60        Method immediateDispatcherMethod = dispatcherClass.getDeclaredMethod("immediate");
61        immediateDispatcherMethod.setAccessible(true);
62  
63        // EventBus constructor that accepts custom executor is not yet part of the public API
64        Constructor<EventBus> factory = EventBus.class.getDeclaredConstructor(
65            String.class, Executor.class, dispatcherClass, SubscriberExceptionHandler.class);
66        factory.setAccessible(true);
67  
68        Object immediateDispatcher = immediateDispatcherMethod.invoke(null);
69  
70        SubscriberExceptionHandler exceptionHandler = new LoggingSubscriberExceptionHandler(
71            LoggerFactory.getLogger(String.format("%s.%s", EventBus.class.getName(), name))
72        );
73  
74        return factory.newInstance(name, executor, immediateDispatcher, exceptionHandler);
75      }
76      catch (Exception e) {
77        Throwables.throwIfUnchecked(e);
78        throw new LinkageError("Unable to create EventBus with custom executor", e);
79      }
80    }
81  }
