13  package org.sonatype.goodies.dropwizard.jersey;
14  
15  import javax.ws.rs.core.Feature;
16  import javax.ws.rs.core.FeatureContext;
17  
18  import org.sonatype.goodies.dropwizard.util.Levels;
19  
20  import ch.qos.logback.classic.Level;
21  import org.glassfish.jersey.logging.LoggingFeature.Verbosity;
22  
23  import static com.google.common.base.Preconditions.checkNotNull;
24  import static com.google.common.base.Preconditions.checkState;
25  
26  /**
27   * Logging feature.
28   *
29   * Wrapper around {@link org.glassfish.jersey.logging.LoggingFeature} to limit configuration types exposed.
30   *
31   * @since 1.0.2
32   */
33  public class LoggingFeature
34      implements Feature
35  {
36    private String logger = getClass().getName();
37  
38    private Level level = Level.TRACE;
39  
40    private Verbosity verbosity = Verbosity.PAYLOAD_ANY;
41  
42    private int maxEntitySize = 1024 * 8;
43  
44    public LoggingFeature(final String logger, final Level level, final Verbosity verbosity, final int maxEntitySize) {
45      this.logger = checkNotNull(logger);
46      this.level = checkNotNull(level);
47      this.verbosity = checkNotNull(verbosity);
48      this.maxEntitySize = maxEntitySize;
49    }
50  
51    public LoggingFeature(final LoggingConfiguration config) {
52      this(config.getLogger(), config.getLevel(), config.getVerbosity(), config.getMaxEntitySize());
53    }
54  
55    public LoggingFeature() {
56      // empty
57    }
58  
59    public String getLogger() {
60      return logger;
61    }
62  
63    public void setLogger(final String logger) {
64      this.logger = logger;
65    }
66  
67    public Level getLevel() {
68      return level;
69    }
70  
71    public void setLevel(final Level level) {
72      this.level = level;
73    }
74  
75    public Verbosity getVerbosity() {
76      return verbosity;
77    }
78  
79    public void setVerbosity(final Verbosity verbosity) {
80      this.verbosity = verbosity;
81    }
82  
83    public int getMaxEntitySize() {
84      return maxEntitySize;
85    }
86  
87    public void setMaxEntitySize(final int maxEntitySize) {
88      this.maxEntitySize = maxEntitySize;
89    }
90  
91    @Override
92    public boolean configure(final FeatureContext context) {
93      checkState(logger != null);
94      checkState(level != null);
95      checkState(verbosity != null);
96      checkState(maxEntitySize > 0);
97  
98      return new org.glassfish.jersey.logging.LoggingFeature(
99          java.util.logging.Logger.getLogger(logger),
100         Levels.convert(level),
101         verbosity,
102         maxEntitySize
103     ).configure(context);
104   }
105 }
