45  /**
46   * Component discovery.
47   *
48   * @since 1.0.0
49   */
50  @Named
51  @Singleton
52  public class ComponentDiscovery
53  {
54    private static final Logger log = LoggerFactory.getLogger(ComponentDiscovery.class);
55  
56    private final BeanLocator locator;
57  
58    @Inject
59    public ComponentDiscovery(final BeanLocator locator) {
60      this.locator = checkNotNull(locator);
61    }
62  
63    /**
64     * Discover components.
65     */
66    public void discover(final Environment environment) {
67      log.debug("Discovering components");
68  
69      detectHealthChecks(environment);
70      detectJaxrsComponents(environment);
71      detectTasks(environment);
72      detectManaged(environment);
73      detectCustomizers(environment);
74  
75      // log order of life-cycle components
76      if (log.isDebugEnabled()) {
77        List<LifeCycle> components = environment.lifecycle().getManagedObjects();
78        log.debug("{} life-cycle components:", components.size());
79        for (LifeCycle component : components) {
80          if (component instanceof JettyManaged) {
81            Managed managed = ((JettyManaged) component).getManaged();
82            log.debug("  {}", managed);
83          }
84        }
85      }
86    }
87  
88    /**
89     * Locate components for given type with optional specific qualifier annotation.
90     */
91    @SuppressWarnings("unchecked")
92    private <Q extends Annotation, C> Iterable<BeanEntry<Q, C>> locate(final Class<C> type, @Nullable final Class<Q> annotation) {
93      List<BeanEntry<Q, C>> result = new ArrayList<>();
94  
95      Key<C> key;
96      if (annotation != null) {
97        key = Key.get(type, annotation);
98      }
99      else {
100       key = Key.get(type);
101     }
102 
103     for (BeanEntry entry : locator.locate(key)) {
104       result.add(entry);
105     }
106 
107     return result;
108   }
109 
110   /**
111    * Locate components for given type.
112    */
113   private <C> Iterable<BeanEntry<Annotation, C>> locate(final Class<C> type) {
114     return locate(type, null);
115   }
116 
117   // TODO: may need a better way to order managed dependencies which are started in order they are added
118   // TODO: ... atm must use @Priority to ensure order, which is a bit obnoxious
119 
120   /**
121    * Detect {@link Managed} components.
122    */
123   private void detectManaged(final Environment environment) {
124     for (BeanEntry<Annotation, Managed> entry : locate(Managed.class)) {
125       Managed component = entry.getValue();
126       environment.lifecycle().manage(component);
127       log.debug("Added managed: {}", component);
128     }
129   }
130 
131   /**
132    * Detect {@link Task} components.
133    */
134   private void detectTasks(final Environment environment) {
135     for (BeanEntry<Annotation, Task> entry : locate(Task.class)) {
136       Task component = entry.getValue();
137       environment.admin().addTask(component);
138       log.debug("Added task: {}", component);
139     }
140   }
141 
142   /**
143    * Detect {@link HealthCheck} components.
144    */
145   private void detectHealthChecks(final Environment environment) {
146     for (BeanEntry<Named, HealthCheck> entry : locate(HealthCheck.class, Named.class)) {
147       String name = entry.getKey().value();
148       HealthCheck component = entry.getValue();
149       environment.healthChecks().register(name, component);
150       log.debug("Added health-check: {} -> {}", name, component);
151     }
152   }
153 
154   /**
155    * Detect JAX-RS components.
156    */
157   private void detectJaxrsComponents(final Environment environment) {
158     // can not use @Provider here, as its not a Guice @Qualifier annotation
159     for (BeanEntry<Annotation, Component> entry : locate(Component.class)) {
160       Component component = entry.getValue();
161       environment.jersey().register(component);
162       log.debug("Added JAX-RS component: {}", component);
163     }
164 
165     for (BeanEntry<Annotation, Resource> entry : locate(Resource.class)) {
166       Resource component = entry.getValue();
167       environment.jersey().register(component);
168       log.debug("Added JAX-RS resource: {}", component);
169     }
170 
171     for (BeanEntry<Annotation, ExceptionMapper> entry : locate(ExceptionMapper.class)) {
172       ExceptionMapper component = entry.getValue();
173       environment.jersey().register(component);
174       log.debug("Added JAX-RS exception-mapper: {}", component);
175     }
176 
177     for (BeanEntry<Annotation, Feature> entry : locate(Feature.class)) {
178       Feature component = entry.getValue();
179       environment.jersey().register(component);
180       log.debug("Added JAX-RS feature: {}", component);
181     }
182 
183     for (BeanEntry<Annotation, DynamicFeature> entry : locate(DynamicFeature.class)) {
184       DynamicFeature component = entry.getValue();
185       environment.jersey().register(component);
186       log.debug("Added JAX-RS dynamic-feature: {}", component);
187     }
188   }
189 
190   /**
191    * Detect and apply {@link EnvironmentCustomizer} components.
192    */
193   private void detectCustomizers(final Environment environment) {
194     for (BeanEntry<Annotation, EnvironmentCustomizer> entry : locate(EnvironmentCustomizer.class)) {
195       EnvironmentCustomizer customizer = entry.getValue();
196       log.debug("Applying customizer: {}", customizer);
197       customizer.customize(environment);
198     }
199   }
