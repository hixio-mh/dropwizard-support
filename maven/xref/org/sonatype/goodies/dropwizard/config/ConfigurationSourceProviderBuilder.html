13  package org.sonatype.goodies.dropwizard.config;
14  
15  import java.util.ArrayList;
16  import java.util.List;
17  
18  import io.dropwizard.configuration.ConfigurationSourceProvider;
19  import io.dropwizard.configuration.SubstitutingSourceProvider;
20  import org.apache.commons.text.StrLookup;
21  import org.apache.commons.text.StrMatcher;
22  import org.apache.commons.text.StrSubstitutor;
23  
24  import static com.google.common.base.Preconditions.checkNotNull;
25  import static com.google.common.base.Preconditions.checkState;
26  
27  /**
28   * {@link ConfigurationSourceProvider} builder.
29   *
30   * @since 1.2.0
31   */
32  public class ConfigurationSourceProviderBuilder
33  {
34    private char escapeChar = StrSubstitutor.DEFAULT_ESCAPE;
35  
36    private StrMatcher prefixMatcher = StrSubstitutor.DEFAULT_PREFIX;
37  
38    private StrMatcher suffixMatcher = StrSubstitutor.DEFAULT_SUFFIX;
39  
40    private StrMatcher valueDelimiterMatcher = StrSubstitutor.DEFAULT_VALUE_DELIMITER;
41  
42    private final List<StrLookup<?>> lookups = new ArrayList<>();
43  
44    private final List<ConfigurationSourceProvider> providers = new ArrayList<>();
45  
46    public ConfigurationSourceProviderBuilder escapeChar(final char escape) {
47      this.escapeChar = escape;
48      return this;
49    }
50  
51    public ConfigurationSourceProviderBuilder prefixMatcher(final StrMatcher matcher) {
52      this.prefixMatcher = checkNotNull(matcher);
53      return this;
54    }
55  
56    public ConfigurationSourceProviderBuilder prefixMatcher(final String token) {
57      checkNotNull(token);
58      return prefixMatcher(StrMatcher.stringMatcher(token));
59    }
60  
61    public ConfigurationSourceProviderBuilder suffixMatcher(final StrMatcher matcher) {
62      this.suffixMatcher = checkNotNull(matcher);
63      return this;
64    }
65  
66    public ConfigurationSourceProviderBuilder suffixMatcher(final String token) {
67      checkNotNull(token);
68      return suffixMatcher(StrMatcher.stringMatcher(token));
69    }
70  
71    public ConfigurationSourceProviderBuilder valueDelimiterMatcher(final StrMatcher matcher) {
72      this.valueDelimiterMatcher = checkNotNull(matcher);
73      return this;
74    }
75  
76    public ConfigurationSourceProviderBuilder valueDelimiterMatcher(final String token) {
77      checkNotNull(token);
78      return valueDelimiterMatcher(StrMatcher.stringMatcher(token));
79    }
80  
81    public ConfigurationSourceProviderBuilder simpleStyle(final char token) {
82      // TODO: verify token is with-in range for yaml
83      escapeChar(token);
84      prefixMatcher(String.format("%s{", token));
85      suffixMatcher("}");
86      valueDelimiterMatcher(":-");
87      return this;
88    }
89  
90    public ConfigurationSourceProviderBuilder lookup(final StrLookup<?> lookup) {
91      checkNotNull(lookup);
92      lookups.add(lookup);
93      return this;
94    }
95  
96    public ConfigurationSourceProviderBuilder provider(final ConfigurationSourceProvider provider) {
97      checkNotNull(provider);
98      providers.add(provider);
99      return this;
100   }
101 
102   public ConfigurationSourceProvider build() {
103     checkState(!lookups.isEmpty(), "At least one lookup must be specified");
104     checkState(!providers.isEmpty(), "At least one provider must be specified");
105 
106     StrLookup lookup;
107     if (lookups.size() == 1) {
108       lookup = lookups.get(0);
109     }
110     else {
111       lookup = new FirstMatchStrLookup(lookups.toArray(new StrLookup[0]));
112     }
113 
114     ConfigurationSourceProvider provider;
115     if (providers.size() == 1) {
116       provider = providers.get(0);
117     }
118     else {
119       provider = new FirstMatchConfigurationSourceProvider(providers.toArray(new ConfigurationSourceProvider[0]));
120     }
121 
122     StrSubstitutor substitutor = new StrSubstitutor(lookup, prefixMatcher, suffixMatcher, escapeChar, valueDelimiterMatcher);
123     return new SubstitutingSourceProvider(provider, substitutor);
124   }
125 }
